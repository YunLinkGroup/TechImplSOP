prometheus.yml

```yaml
# 全局配置
global:
  scrape_interval: 15s # 每次抓取间隔 15 秒；默认 1 分钟
  evaluation_interval: 15s # 每 15 秒评估一次规则；默认 1 分钟。
  # scrape_timeout 采用全局默认值 10 秒。

# 抓取配置，此处样例为抓取 Prometheus 自身指标 endpoint
scrape_configs:
  # job_name 会作为标签 job=<job_name> 被添加到此配置中抓取的任何时间序列中。
  - job_name: "prometheus"

    # 指标路径 metrics_path 默认为 /metrics
    # 协议 scheme 默认为 http

    static_configs:
      - targets: [ "localhost:9090" ]
        # 自定义标签：app=“prometheus”将被添加到此配置中抓取的任何时间序列中。
        labels:
          app: "prometheus"
  - job_name: "node"

    # 覆盖全局默认值，并每5秒从该作业中抓取目标。
    scrape_interval: 5s

    static_configs:
      # 生产实例
      - targets: [ "localhost:20001", "localhost:20002" ]
        labels:
          group: "production"
          # 金丝雀实例
      - targets: [ "localhost:20003" ]
        labels:
          group: "canary"

# 加载规则文件，并按全局 evaluation_interval 定期评估。
rule_files:
  - "rules/*.yml"
# - "first_rules.yml"
# - "second_rules.yml"

# Alertmanager 配置
alerting:
  alertmanagers:
    - static_configs:
        - targets:
            - "localhost:9093"
```

rule/node.yml

```yaml
# node 告警规则


# Prometheus 告警规则与 Alertmanager 配置示例
# 用途：监控节点 CPU 使用率，当系统态 CPU 使用率 > 80 % 并持续 1 min 时触发告警，并将告警推送至 Alertmanager。

groups:
# 规则组名称
- name: node.rules        # 规则组名称，便于在 Prometheus 界面中识别与管理
  rules:
  - alert: HighCPUUsage      # 告警名称，需保持唯一性
    # 表达式说明：
    # 1. node_cpu_seconds_total{mode="system"}  仅取“系统态”CPU 时间
    # 2. rate(...[5m])                           计算 5 min 内每秒平均增长率，即系统态 CPU 使用率
    # 3. sum by (instance) (...)                 按实例汇总，支持多核/多 NUMA 场景
    # 4. * 100                                   转换成百分比
    # 5. > 80                                    阈值判断，超过 80 % 即满足触发条件
    expr: sum by (instance) (rate(node_cpu_seconds_total[5m])) * 100 > 80
    for: 1m                   # 持续 1 min 才生效，避免瞬时抖动产生误报
    labels:
      severity: page          # 自定义标签，可用于 Alertmanager 路由、抑制、合并等策略
    annotations:
      # 简短摘要，$labels.instance 会在告警触发时被替换成实际实例地址
      summary: "High CPU usage detected on {{ $labels.instance }}"
      # 详细描述，$value 会被替换成当前计算出的 CPU 使用率百分比
      description: "{{ $labels.instance }} of job node-exporter has high CPU usage ({{ $value }}%)"
```
