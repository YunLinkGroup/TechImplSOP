rabbitmq-local.yaml

```yaml
# 0. 命名空间
apiVersion: v1
kind: Namespace
metadata:
  name: mid
---
# 1. 本地 PV
apiVersion: v1
kind: PersistentVolume
metadata:
  name: rabbitmq-local-pv
spec:
  capacity:
    storage: 10Gi
  accessModes: ["ReadWriteOnce"]
  persistentVolumeReclaimPolicy: Retain
  storageClassName: local-rabbitmq
  local:
    path: /storage/rabbitmq
  nodeAffinity:
    required:
      nodeSelectorTerms:
      - matchExpressions:
        - key: kubernetes.io/hostname
          operator: In
          values: ["izrj943cxeelu84r2mp9adz"]   # 改成你的节点名
---
# 2. PVC
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: rabbitmq-data
  namespace: mid
spec:
  accessModes: ["ReadWriteOnce"]
  storageClassName: local-rabbitmq
  resources:
    requests:
      storage: 10Gi
---
# 3. Secret（admin 用户密码）
apiVersion: v1
kind: Secret
metadata:
  name: rabbitmq-secret
  namespace: mid
type: Opaque
stringData:
  password: "rAb123456"

---
# 5. Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: rabbitmq
  namespace: mid
spec:
  replicas: 1
  selector:
    matchLabels: { app: rabbitmq }
  template:
    metadata:
      labels: { app: rabbitmq }
    spec:
      nodeSelector:
        kubernetes.io/hostname: izrj943cxeelu84r2mp9adz
      containers:
      - name: rabbitmq
        image: rabbitmq:3.13-management-alpine   # 已带 management 插件
        ports:
        - containerPort: 5672
          name: amqp
        - containerPort: 15672
          name: management
        env:
        - name: RABBITMQ_DEFAULT_USER
          value: "admin"
        - name: RABBITMQ_DEFAULT_PASS
          valueFrom:
            secretKeyRef:
              name: rabbitmq-secret
              key: password
        volumeMounts:
        - name: data
          mountPath: /var/lib/rabbitmq
        livenessProbe:
          exec:
            command: ["rabbitmq-diagnostics", "ping"]
          initialDelaySeconds: 60
          periodSeconds: 30
        readinessProbe:
          exec:
            command: ["rabbitmq-diagnostics", "ping"]
          initialDelaySeconds: 20
          periodSeconds: 10
        resources:
          requests:
            cpu: "500m"
            memory: "512Mi"
          limits:
            cpu: "1"
            memory: "1Gi"
      volumes:
      - name: data
        persistentVolumeClaim:
          claimName: rabbitmq-data
          
          
---
# 4. Service
apiVersion: v1
kind: Service
metadata:
  name: rabbitmq
  namespace: mid
spec:
  type: NodePort
  ports:
  - port: 5672
    targetPort: 5672
    nodePort: 30672
    name: amqp
  - port: 15672
    targetPort: 15672
    nodePort: 30673
    name: management
  selector: { app: rabbitmq }
```

```bash
kubectl apply -f rabbitmq-local.yaml
```
