  name: grafana-local-pv

  accessModes: [ReadWriteOnce]
  persistentVolumeReclaimPolicy: Retain
  storageClassName: local-mon
  local: { path: /storage/grafana }
  nodeAffinity:
    required:
      nodeSelectorTerms:
      - matchExpressions:
        - key: kubernetes.io/hostname
          operator: In
          values: ["izrj943cxeelu84r2mp9adz"]

apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: grafana-data
  namespace: mon
spec:
  accessModes: [ReadWriteOnce]
  storageClassName: local-mon
  resources: { requests: { storage: 5Gi } }
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: grafana-datasources
  namespace: mon
data:
  prometheus.yaml: |-
    apiVersion: 1
    datasources:
    - name: Prometheus
      type: prometheus
      url: http://prometheus:9090
      access: proxy
      isDefault: true

apiVersion: apps/v1
kind: Deployment
metadata:
  name: grafana
  namespace: mon

  replicas: 1
  selector:
    matchLabels: { app: grafana }
  template:
    metadata:
      labels: { app: grafana }
    spec:
      nodeSelector: { kubernetes.io/hostname: izrj943cxeelu84r2mp9adz }
      initContainers:
      - name: fix-perm

      - name: grafana
        image: grafana/grafana:10.2.2
        ports:
        - containerPort: 3000
        env:
        - name: GF_SECURITY_ADMIN_PASSWORD
          value: "gF123456"

        - name: data
          mountPath: /var/lib/grafana
        - name: datasources
          mountPath: /etc/grafana/provisioning/datasources

          requests: { cpu: "100m", memory: "256Mi" }
          limits: { cpu: "500m", memory: "512Mi" }

      - name: datasources
        configMap: { name: grafana-datasources }
