放在 worker 2

postgresql-16-local.yaml

```yaml
# 1. 本地 PV
apiVersion: v1
kind: PersistentVolume
metadata:
  name: pg-local-pv
spec:
  capacity:
    storage: 20Gi
  accessModes: ["ReadWriteOnce"]
  persistentVolumeReclaimPolicy: Retain
  storageClassName: local-pg
  local:
    path: /storage/pg
  nodeAffinity:
    required:
      nodeSelectorTerms:
      - matchExpressions:
        - key: kubernetes.io/hostname
          operator: In
          values: ["izrj943cxeelu84r2mp9adz"]   # 改成你的节点名
---
# 2. PVC
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: pg-data
  namespace: mid
spec:
  accessModes: ["ReadWriteOnce"]
  storageClassName: local-pg
  resources:
    requests:
      storage: 20Gi
---
# 3. Secret（postgres 用户密码）
apiVersion: v1
kind: Secret
metadata:
  name: pg-secret
  namespace: mid
type: Opaque
stringData:
  password: "pG123456_"
---
# 4. Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: postgres
  namespace: mid
spec:
  replicas: 1
  selector:
    matchLabels: { app: postgres }
  template:
    metadata:
      labels: { app: postgres }
    spec:
      nodeSelector:
        kubernetes.io/hostname: izrj943cxeelu84r2mp9adz
      containers:
      - name: postgres
        image: postgres:16-alpine
        ports:
        - containerPort: 5432
          name: postgres
        env:
        - name: POSTGRES_PASSWORD
          valueFrom:
            secretKeyRef:
              name: pg-secret
              key: password
        - name: POSTGRES_DB
          value: postgres
        volumeMounts:
        - name: data
          mountPath: /var/lib/postgresql/data
          subPath: pgdata     # 避免挂载点 lost+found
        livenessProbe:
          exec:
            command:
            - pg_isready
            - -U
            - postgres
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          exec:
            command:
            - pg_isready
            - -U
            - postgres
          initialDelaySeconds: 5
          periodSeconds: 5
        resources:
          requests:
            cpu: "500m"
            memory: "1Gi"
          limits:
            cpu: "1"
            memory: "2Gi"
      volumes:
      - name: data
        persistentVolumeClaim:
          claimName: pg-data
---
# 5. Service
apiVersion: v1
kind: Service
metadata:
  name: postgres
  namespace: mid
spec:
  type: NodePort
  ports:
  - port: 5432
    targetPort: 5432
    nodePort: 30432
    name: postgres
  selector: { app: postgres }
```

```bash
kubectl apply -f postgresql-16-local.yaml
kubectl get pod -n mid
```
