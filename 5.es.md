elasticsearch 可以简写为 es

```bash
sudo mkdir -p /storage/es
sudo chown -R 1000:1000 /storage/es     # ES 容器默认 UID 1000
```

放在 worker 1

es-local.yaml

```yaml
# 1. 本地持久卷
apiVersion: v1
kind: PersistentVolume
metadata:
  name: es-local-pv
spec:
  capacity:
    storage: 20Gi
  accessModes: [ "ReadWriteOnce" ]
  persistentVolumeReclaimPolicy: Retain
  storageClassName: local-es
  local:
    path: /storage/es  # 节点本地路径，需提前在节点建好 /storage/es 目录
  nodeAffinity:
    required:
      nodeSelectorTerms:
        - matchExpressions:
            - key: kubernetes.io/hostname
              operator: In
              values:
                - izrj943cxeelu84r2mp9acz # 此处改为指定的 k8s 节点名
---
# 2. PVC
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: es-data
  namespace: mid
spec:
  accessModes: [ "ReadWriteOnce" ]
  storageClassName: local-es
  resources:
    requests:
      storage: 20Gi
---
# 3. ES 配置 ConfigMap
apiVersion: v1
kind: ConfigMap
metadata:
  name: es-config
  namespace: mid
data:
  elasticsearch.yml: |
    cluster.name: es-single
    node.name: es-0
    path.data: /usr/share/elasticsearch/data
    path.logs: /usr/share/elasticsearch/logs
    network.host: 0.0.0.0
    http.port: 9200
    transport.port: 9300
    discovery.type: single-node
    xpack.security.enabled: true
    xpack.security.http.ssl.enabled: false   # 集群内部不用 HTTPS，省事
    xpack.security.transport.ssl.enabled: false
---
# 4. Deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: es
  namespace: mid
spec:
  replicas: 1
  selector:
    matchLabels: { app: es }
  template:
    metadata:
      labels: { app: es }
    spec:
      nodeSelector:
        kubernetes.io/hostname: izrj943cxeelu84r2mp9acz
      initContainers:
        - name: fix-perm
          image: busybox:1.36
          command: [ 'sh','-c','chown -R 1000:1000 /usr/share/elasticsearch/data' ]
          volumeMounts:
            - name: data
              mountPath: /usr/share/elasticsearch/data
      containers:
        - name: elasticsearch
          image: docker.elastic.co/elasticsearch/elasticsearch:8.17.3
          ports:
            - containerPort: 9200
              name: http
            - containerPort: 9300
              name: transport
          env:
            - name: ES_JAVA_OPTS
              value: "-Xms1g -Xmx1g"
            - name: ELASTIC_PASSWORD
              value: "eS123456_" # 超级用户 elastic 的密码
          volumeMounts:
            - name: config
              mountPath: /usr/share/elasticsearch/config/elasticsearch.yml
              subPath: elasticsearch.yml
            - name: data
              mountPath: /usr/share/elasticsearch/data
          resources:
            requests:
              cpu: "500m"
              memory: "1Gi"
            limits:
              cpu: "2"
              memory: "4Gi"
          livenessProbe:
            tcpSocket: { port: 9200 }
            initialDelaySeconds: 60
            periodSeconds: 30
          readinessProbe:
            httpGet:
              path: /_cluster/health
              port: 9200
              httpHeaders:
                - name: Authorization
                  value: "Basic ZWxhc3RpYzplUzEyMzQ1Nl8="  # elastic:eS123456_ base64
            initialDelaySeconds: 30
            periodSeconds: 15
      volumes:
        - name: config
          configMap:
            name: es-config
        - name: data
          persistentVolumeClaim:
            claimName: es-data
---
# 5. Service
apiVersion: v1
kind: Service
metadata:
  name: es
  namespace: mid
spec:
  type: NodePort
  ports:
    - port: 9200
      targetPort: 9200
      nodePort: 30020
      name: http
    - port: 9300
      targetPort: 9300
      nodePort: 30021
      name: transport
  selector: { app: es }
```



```shell
kubectl apply -f es-local.yaml
kubectl -n mid get pod
```
